function anagrams(word, words) {
  var w1, w2, m =[]
  for(let i =0; i< words.length; i++){
    w1 = word.split('').sort().join('')
    w2 = words[i].split('').sort().join('')
    if(w1 === w2)  m.push(words[i])
  }
  return m
}
